\section*{(b)}
\begin{align*}
    \frac{\partial J_{\mathrm{naive-softmax}}(v_c, o, U)}{\partial v_{c}} 
    &= - \frac{\partial }{\partial v_{c}} \log \mathbb{P}(O=o \mid C=c) \\
    &= - \frac{\partial }{\partial v_{c}} \log \frac{\exp(u_{o}^{\top}v_{c})}{\sum_{w \in \mathrm{Vocab}} \exp(u_{w}^{\top}v_{c})} \\
    &= - \frac{\partial }{\partial v_{c}} \left[ \log \exp(u_{o}^{\top}v_{c}) - \log \sum_{w \in \mathrm{Vocab}} \exp(u_{w}^{\top}v_{c}) \right] \\
    &= - \frac{\partial }{\partial v_{c}} \left[ u_{o}^{\top}v_{c} - \log \sum_{w \in \mathrm{Vocab}} \exp(u_{w}^{\top}v_{c}) \right] \\
    &= -u_{o} + \frac{\partial }{\partial v_{c}} \left[  \log \sum_{w \in \mathrm{Vocab}} \exp(u_{w}^{\top}v_{c}) \right] \\
    &= -u_{o} + \frac{1}{\sum_{w \in \mathrm{Vocab}} \exp(u_{w}^{\top}v_{c})} \frac{\partial }{\partial v_{c}} \left[ \sum_{w \in \mathrm{Vocab}} \exp(u_{w}^{\top}v_{c}) \right] \\
    &= -u_{o} + \frac{1}{\sum_{w \in \mathrm{Vocab}} \exp(u_{w}^{\top}v_{c})}  \left[ \sum_{w \in \mathrm{Vocab}} \exp(u_{w}^{\top}v_{c}) \frac{\partial }{\partial v_{c}} (u_{w}^{\top}v_{c}) \right] \\
    &= -u_{o} + \frac{1}{\sum_{w \in \mathrm{Vocab}} \exp(u_{w}^{\top}v_{c})}  \left[ \sum_{w \in \mathrm{Vocab}} \exp(u_{w}^{\top}v_{c}) u_{w} \right] \\
    &= -u_{o} +  \sum_{w \in \mathrm{Vocab}} \frac{\exp(u_{w}^{\top}v_{c}) u_{w}}{\sum_{x \in \mathrm{Vocab}} \exp(u_{x}^{\top}v_{c})}\\
    &= -u_{o} + \sum_{w \in \mathrm{Vocab}}\mathbb{P}( O=w \mid C=c )u_{w}\\
    &= -u_{o} + \sum_{w \in \mathrm{Vocab}}\hat{y}_{w}u_{w}\\
    &= -y_{o}u_{o} + \sum_{w \in \mathrm{Vocab}}\hat{y}_{w}u_{w}\\
    &= - \sum_{x \in \mathrm{Vocab}}y_{x}u_{x} + \sum_{w \in \mathrm{Vocab}}\hat{y}_{w}u_{w}\\
    &= \sum_{x \in \mathrm{Vocab}}(\hat{y}_x - y_{x})u_{x} \\
    &= U(\hat{y} - y)
\end{align*}

\section*{(c)}

\textbf{Case 1: $w \neq o$}
\begin{align*}
    \frac{\partial J_{\mathrm{naive-softmax}}(v_c, o, U)}{\partial u_{w}}
    &= - \frac{\partial }{\partial u_{w}} \left[ u_{o}^{\top}v_{c} - \log \sum_{x \in \mathrm{Vocab}} \exp(u_{x}^{\top}v_{c}) \right] \\
    &= \frac{\partial }{ \partial u_{w}} \left[\log \sum_{x \in \mathrm{Vocab}} \exp(u_{x}^{\top} v_{c})\right] \\
    &= \frac{1}{\sum_{x \in \mathrm{Vocab}} \exp(u_{x}^{\top} v_{c})} \frac{\partial }{ \partial u_{w}} \left[ \sum_{x \in \mathrm{Vocab}} \exp(u_{x}^{\top} v_{c})\right] \\
    &= \frac{1}{\sum_{x \in \mathrm{Vocab}} \exp(u_{x}^{\top} v_{c})}  \left[ \sum_{x \in \mathrm{Vocab}} \frac{\partial }{ \partial u_{w}} \exp(u_{x}^{\top} v_{c})\right] \\
    &= \frac{1}{\sum_{x \in \mathrm{Vocab}} \exp(u_{x}^{\top} v_{c})}  \left[ \sum_{x \in \mathrm{Vocab}} \exp(u_{x}^{\top} v_{c}) \frac{\partial }{ \partial u_{w}} (u_{x}^{\top} v_{c})\right]\\
    &= \frac{1}{\sum_{x \in \mathrm{Vocab}} \exp(u_{x}^{\top} v_{c})}  \exp(u_{w}^{\top} v_{c}) \frac{\partial }{ \partial u_{w}} (u_{w}^{\top} v_{c}) \\
    &= \frac{\exp(u_{w}^{\top} v_{c}) }{\sum_{x \in \mathrm{Vocab}} \exp(u_{x}^{\top} v_{c})} v_{c} \\
    &= \hat{y}_{w} v_{c}
\end{align*}
\textbf{Case 2: $w = o$}
\begin{align*}
    \frac{\partial J_{\mathrm{naive-softmax}}(v_c, o, U)}{\partial u_{w}}
    &= - \frac{\partial }{\partial u_{w}} \left[ u_{o}^{\top}v_{c} - \log \sum_{x \in \mathrm{Vocab}} \exp(u_{x}^{\top}v_{c}) \right] \\
    &= - \frac{\partial }{\partial u_{w}} ( u_{o}^{\top}v_{c}) +  \hat{y}_{w} v_{c} \\
    &= -v_{c} +  \hat{y}_{w} v_{c} \\
    &= ( \hat{y}_{w} - 1)v_{c}
\end{align*}

\medskip
\noindent Thus,
\[
    \frac{\partial J_{\mathrm{naive-softmax}}}{\partial u_w} =
    \begin{cases}
        (\hat{y}_o - 1) \, v_c, & w = o \\
        \hat{y}_w \, v_c,       & w \neq o
    \end{cases}
\]

\noindent Equivalently, since $ y_{o} = 1 $ and $ y_{w} = 0$ ($w \neq o$),
this can be written as: 
\[
    \frac{\partial J_{\mathrm{naive-softmax}}}{\partial u_w} = (\hat{y}_w - y_{w}) v_c.
\]
